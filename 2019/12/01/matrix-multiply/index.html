矩阵乘法
java 矩阵乘法
并发编程学习
发布日期:   2019-12-01
阅读次数:  

矩阵乘法是针对矩阵做的基本运算之一，也是并发和并行编程课程中常采用的经典问题。如果你有一个 m 行 n 列的矩阵A，和另一个 n 行 s 列的矩阵 B，那么可以将两个矩阵相乘得到一个 m 行 s 列的矩阵 C。简单的说就是，两个矩阵，内标同可乘，外标决定型。

首先创建一个名为 MatrixGenerator 的类，用来生成矩阵，里面只有一个方法。

public static int[][] generate (int rows, int columns) {
        int[][] ret = new int[rows][columns];
        Random random = new Random();
        for (int i=0; i<rows; i++) {
            for (int j = 0; j < columns; j++) {
                ret[i][j] = random.nextInt(10);        // 每个元素大小为[0, 10)
            }
        }
        return ret;
    }

1、串行版本

首先创建一个名为SerialMultiplier，里面提供一个用来对矩阵做相乘运算的方法

public static void multiply(int[][] matrix1, int[][] matrix2, int[][] result) {
        int rows1 = matrix1.length;
        int columns1 = matrix1[0].length;
        int columns2 = matrix2[0].length;
        for (int i = 0; i < rows1; i++) {
            for (int j = 0; j < columns2; j++) {
                result[i][j] = 0;
                for (int k = 0; k < columns1; k++) {
                    result[i][j] += matrix1[i][k] * matrix2[k][j];
                }
            }
        }
    }

接着写一个含有主方法的类，命名为SerialMain。

public static void main(String[] args) {
        int matrix1[][] = MatrixGenerator.generate(500, 300);
        int matrix2[][] = MatrixGenerator.generate(300, 400);
        int resultSerial[][]= new int[matrix1.length][matrix2[0].length];
        long start = System.currentTimeMillis();    // 开始时间
        SerialMultiplier.multiply(matrix1, matrix2, resultSerial);
        long end = System.currentTimeMillis();        // 结束时间
        System.out.println("耗时：" + (end - start) + "ms"); //耗时：202 ~ 375ms
    }

2、矩阵的每个元素作为一个单独的线程

首先创建一个名为IndividualMultiplierTask的类，用以实现每个Thread。

public class IndividualMultiplierTask implements Runnable {
    private final int[][] result;
    private final int[][] matrix1;
    private final int[][] matrix2;
    private final int row;
    private final int column;

    public IndividualMultiplierTask(int[][] result, int[][]
            matrix1, int[][] matrix2, int row, int cloumn) {
        this.result = result;
        this.matrix1 = matrix1;
        this.matrix2 = matrix2;
        this.row = row;
        this.column = cloumn;
    }

    @Override
    public void run() {
        result[row][column] = 0;
        for (int k = 0; k < matrix1[row].length; k++) {
            result[row][column] += matrix1[row][k] * matrix2[k][column];
        }
    }
}

接着新建一个用于计算的类ParallelIndividualMultiplier，此处为了防止同时启动的线程数量过多，所以将以 10 个线程一组的形式启动线程。

public class ParallelIndividualMultiplier {

    public static void multiply(int[][] matrix1, int[][] matrix2, int[][] result) {
        List<Thread> threads = new ArrayList<>();
        int rows = matrix1.length;
        int column = matrix2[0].length;
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < column; j++) {
                IndividualMultiplierTask task = new IndividualMultiplierTask(result, matrix1, matrix2, i, j);
                Thread thread = new Thread(task);
                thread.start();
                threads.add(thread);
                if (threads.size() % 10 == 0) {
                    waitForThreads(threads);
                }
            }
        }
    }

    private static void waitForThreads(List<Thread> threads) {
        for (Thread thread : threads) {
            try {
                thread.join();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }
        threads.clear();
    }
}

最后创建一个含有main函数的类ParallelIndividualMain

public static void main(String[] args) {
        int matrix1[][] = MatrixGenerator.generate(500, 300);
        int matrix2[][] = MatrixGenerator.generate(300, 400);
        int result[][]= new int[matrix1.length][matrix2[0].length];
        long start = System.currentTimeMillis();
        ParallelIndividualMultiplier.multiply(matrix1, matrix2, result);
        long end = System.currentTimeMillis();
        System.out.println("耗时：" + (end - start) + "ms"); //10个线程一组 耗时：22147ms
    }

3、矩阵每一行对应一个线程

一个实现Runable的类RowMultiplierTask

public class RowMultiplierTask implements Runnable{
    private final int[][] result;
    private final int[][] matrix1;
    private final int[][] matrix2;
    private final int row;
    public RowMultiplierTask(int[][] matrix1, int[][] matrix2, int[][] result, int row){
        this.result = result;
        this.matrix1 = matrix1;
        this.matrix2 = matrix2;
        this.row = row;
    }

    @Override
    public void run() {
        for (int i = 0; i < matrix2[0].length; i++) {
            result[row][i] = 0;
            for (int j = 0; j < matrix1[0].length; j++) {
                result[row][i] += matrix1[row][j] * matrix2[j][i];
            }
        }
    }
}

接着创建一个用于计算的类ParallelRowMultiplier，此处也将已 10 个线程一组的形式启动线程。想练习一下java 8的新特性。

public static void multiply(int[][] matrix1, int[][] matrix2, int[][] result){
        List<Thread> list = new ArrayList<>();
        for (int i = 0; i < matrix1.length; i++) {
            RowMultiplierTask rowMultiplierTask = new RowMultiplierTask(matrix1, matrix2, result, i);
            Thread thread = new Thread(rowMultiplierTask);
            thread.start();
            list.add(thread);
            if (list.size() % 10 == 0){
                waitForThreads(list);
            }
        }
    }

    private static void waitForThreads(List<Thread> list){
        list.stream().forEach(t->{
            try {
                t.join();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        });
        list.clear();
    }

最后创建一个含有main函数的类ParallelRowMain

public static void main(String[] args) {
        int matrix1[][] = MatrixGenerator.generate(500, 300);
        int matrix2[][] = MatrixGenerator.generate(300, 400);
        int result[][]= new int[matrix1.length][matrix2[0].length];
        long start = System.currentTimeMillis();
        ParallelRowMultiplier.multiply(matrix1, matrix2, result);
        long end = System.currentTimeMillis();
        System.out.println("耗时：" + (end - start) + "ms");   //耗时：210 ~ 300ms
    }

4、线程的数量由本机处理器决定

一个实现Runable的类GroupMultiplierTask

public class GroupMultiplierTask implements Runnable {

    private final int[][] matrix1;
    private final int[][] matrix2;
    private final int[][] result;
    private final int startIndex;    // 每个处理器需要计算的开始行索引
    private final int endIndex;        // 每个处理器需要计算的结束行索引

    public GroupMultiplierTask(int[][] matrix1, int[][] matrix2, int[][] result,
                               int startIndex, int endIndex){
        this.matrix1 = matrix1;
        this.matrix2 = matrix2;
        this.result = result;
        this.startIndex = startIndex;
        this.endIndex = endIndex;
    }

    @Override
    public void run() {
        for (int i = startIndex; i < endIndex; i++) {
            for (int j = 0; j < matrix2[0].length; j++) {
                result[i][j] = 0;
                for (int k = 0; k < matrix1[i].length; k++) {
                    result[i][j] += matrix1[i][k] * matrix2[k][j];
                }
            }
        }
    }
}

接着创建一个用于计算的类ParallelGroupMutiplie

public class ParallelGroupMutiplier {

    public static void multiply(int[][] matrix1, int[][] matrix2, int[][] result) {
        int rows1 = matrix1.length;
        int processors = Runtime.getRuntime().availableProcessors();    // 获取本机可用处理器数量
        int startIndex, endIndex, step;
        step = rows1 / processors;
        startIndex = 0;
        endIndex = step;
        for (int i = 0; i < processors; i++) {    // 将决定每个处理器计算哪几行
            GroupMultiplierTask groupMultiplierTask = new GroupMultiplierTask(matrix1, matrix2, result, startIndex, endIndex);
            Thread t = new Thread(groupMultiplierTask);
            t.start();
            startIndex = endIndex;
            endIndex = (i == processors - 1 ? rows1 : endIndex + step);
        }
    }
}

最后，还是一个含有main函数的类ParallelGroupMain

public static void main(String[] args) {
        int matrix1[][] = MatrixGenerator.generate(500, 300);
        int matrix2[][] = MatrixGenerator.generate(300, 400);
        int result[][]= new int[matrix1.length][matrix2[0].length];
        long start = System.currentTimeMillis();
        ParallelGroupMutiplier.multiply(matrix1, matrix2, result);
        long end = System.currentTimeMillis();
        System.out.println("耗时：" + (end - start) + "ms");   // 2 ~ 6ms
    }

5、总结

按分组处理的并行版和按行处理的并行版得到了最佳结果，而按个体即每个元素处理的并行版得到的结果最差。这个例子告诉我们，开发一个并发应用程序时必须非常小心。如果没有选择良好的解决方案，那么性能表现会很糟糕。
